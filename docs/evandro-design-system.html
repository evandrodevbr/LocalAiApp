Motion System

Hover nos cards para ver as animações. Motion não é decoração — tem semântica.

Fade · Entrada/Saída

Slide · Navegação

motion.config.ts
// Curvas de easing do sistema — nunca use ease padrão do CSS
export const easing = {
  // Entradas de UI — sensação de gravidade natural
  out:    'cubic-bezier(.16, 1, .3, 1)',
  // Saídas — rápido para não bloquear o usuário
  in:     'cubic-bezier(.4, 0, 1, 1)',
  // Micro-interações críticas (botão, link)
  spring: 'cubic-bezier(.34, 1.56, .64, 1)',
}

export const duration = {
  instant: 100,  // hover state
  fast:    200,  // botões, toggles
  normal:  350,  // modais, drawers
  slow:    600,  // page transitions
}
